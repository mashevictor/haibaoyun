首页「实际视频」像素大小

这里要的是正在播放的 <video> 元素在页面上的显示尺寸（不是外层容器），以及视频文件本身的分辨率。

—
操作：在浏览器中打开首页，等第一段视频开始播放后，按 F12 打开开发者工具，切到 Console（控制台），粘贴下面整段代码后回车： 
(function(){
  var v = document.querySelector('#hero-sequencer-inner .video-layer:not(.invisible) .hero-center-video') || document.querySelector('#hero-sequencer-inner .hero-center-video');
  if (!v) { console.log('未找到视频元素'); return; }
  var r = v.getBoundingClientRect();
  console.log('--- 实际 video 元素在页面上的像素 ---');
  console.log('width: ' + Math.round(r.width) + ' px, height: ' + Math.round(r.height) + ' px');
  console.log('left: ' + Math.round(r.left) + ' px, top: ' + Math.round(r.top) + ' px');
  if (v.videoWidth) {
    console.log('--- 视频文件本身分辨率 ---');
    console.log('videoWidth: ' + v.videoWidth + ' px, videoHeight: ' + v.videoHeight + ' px');
  }
  return { display: { width: r.width, height: r.height, left: r.left, top: r.top }, source: v.videoWidth ? { width: v.videoWidth, height: v.videoHeight } : null };
})();
控制台会打印：① 实际视频在页面上的宽高、位置；② 视频源分辨率（videoWidth / videoHeight）。
